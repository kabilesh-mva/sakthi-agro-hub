-- Create reviews table CREATE TABLE IF NOT EXISTS public.reviews ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), name TEXT NOT NULL, rating INTEGER NOT NULL CHECK (rating >= 1 AND rating <= 5), comment TEXT NOT NULL, created_at TIMESTAMPTZ NOT NULL DEFAULT now() ); -- Enable Row Level Security ALTER TABLE public.reviews ENABLE ROW LEVEL SECURITY; -- Drop existing policies DROP POLICY IF EXISTS "Allow public read" ON public.reviews; DROP POLICY IF EXISTS "Allow public insert" ON public.reviews; -- Create policies CREATE POLICY "Allow public read" ON public.reviews FOR SELECT TO authenticated, anon USING (true); CREATE POLICY "Allow public insert" ON public.reviews FOR INSERT TO authenticated, anon WITH CHECK (true); -- ‚úÖ Reviews table ready!
